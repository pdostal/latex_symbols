	\section*{Diferenciální počet}
	\begin{multicols}{3}
		\begin{itemize}
			\item $(konst.)' = 0$
			\item $(x^{a})' = a \cdot x^{a-1}$
			\item $(a^{x})' = a^{x} \cdot \ln(a)$
			\item $\log_{a}(x)' = \frac{1}{x \cdot \ln(a)}$
			\item $(e^{x})' = e^{x}$
			\item $(\ln{x})' = \frac{1}{x}$
			\item $(u \pm v)' = u' \pm v'$
			\item $[f(g(x))]' = f'(g(x)) \cdot g'(x)$
		\columnbreak
			\item $\sin(x)' = \cos(x)$
			\item $\cos(x)' = -\sin(x)$
			\item $\tan(x)' = \frac{1}{\cos^2{x}}$
			\item $\cot(x)' = \frac{1}{\sin^2{x}}$
			\item $(x)' = 1$
			\item $\left(\frac{1}{x}\right)' = -\frac{1}{x^{2}}$
			\item $(u \cdot v)' = u' \cdot v + v \cdot u'$
			\item $f(x)^{g(x)} = e^{g(x) \cdot \ln{g(x)}}$
		\columnbreak
			\item $\arcsin(x)' = \frac{1}{\sqrt{1-x^2}}$
			\item $\arccos(x)' = -\frac{1}{\sqrt{1-x^2}}$
			\item $\arctan(x)' = \frac{1}{1+x^2}$
			\item $arccot(x)' = -\frac{1}{1+x^2}$
			\item $\left(\sqrt{x}\right)' = \frac{1}{2 \cdot \sqrt{x}}$
			\item $(\log{x})' = \frac{1}{x \cdot \ln{10}}$
			\item $\left(\frac{u}{v}\right)' = \frac{u' \cdot v - u \cdot v'}{v^{2}}$
			\item $(k \cdot f(x))' = k \cdot f'(x)$
		\end{itemize}
	\end{multicols}

			\item $(a-b)^3 =a^3 b^0 -3a^2 b^1 +3a^1 b^2 - a^0 b^3$
			\item $a^3 + b^3 =(a+b)(a^2 - ab + b^2 )$

	\subsection{Mocniny}
	\begin{math}
	a^{-n}=\frac{1}{a^n}\\
	a^{\frac{m}{n}}=\sqrt[n]{a^m}\\
	(a*b)^n =a^n *b^n\\
	a^m * a^n = a^{m+n}\\
	\frac{a^m}{a^n}=a^{m-n}\\
	(a^m)^n = a^{m*n}\\
	\left(\frac{a}{b}\right)^n = \frac{a^m}{a^n}
	\end{math}
