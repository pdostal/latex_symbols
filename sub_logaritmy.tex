\section*{Logaritmy}
\begin{multicols}{2}
  \begin{itemize}
    \item $y=\log_a{x} \Leftrightarrow a^y = x$
    \item $\log_a{\{x_1*x_2\}}=log_a{x_1} + log_a{x_2}$
  \columnbreak
    \item $\log_a{\left(\frac{x_1}{x_2}\right)}=log_a{x_1} - log_a{x_2}$
    \item $\log_a{x^n}=n*\log_a{x}$
  \end{itemize}
\end{multicols}
