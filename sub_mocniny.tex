\section*{Mocniny}
\begin{multicols}{2}
  \begin{itemize}
    \item $a^{-n}=\frac{1}{a^n}$
    \item $a^{\frac{m}{n}}=\sqrt[n]{a^m}$
    \item $(a*b)^n =a^n *b^n$
    \item $a^m * a^n = a^{m+n}$
  \columnbreak
    \item $\frac{a^m}{a^n}=a^{m-n}$
    \item $(a^m)^n = a^{m*n}$
    \item $\left(\frac{a}{b}\right)^m = \frac{a^m}{a^m}$
  \end{itemize}
\end{multicols}
